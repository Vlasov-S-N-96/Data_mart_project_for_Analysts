🎯 Цель проекта

Предоставить аналитикам агрегированные метрики по карточкам товаров, популярным категориям и прочим параметрам на маркетплейсе. Демонстрация совместной работы DWH (Greenplum), Data Lake (Spark) и Data Orchestrator (Airflow).

🧩 Конечная цель витрины: создание инструментов для упрощения аналитики данных по товарам.

🔥 Разработка Spark Job

Чтение из S3, вычисление метрик: returned_items_count, potential_revenue, total_revenue, avg_daily_sales, days_to_sold, item_rate_percent.

# Фрагмент PySpark: агрегации и оконные функции
item_df = spark.read.parquet("s3a://startde-raw/raw_items")

item_df = item_df.withColumn("returned_items_count",
    F.round((F.col("ordered_items_count") * (1 - (F.col("avg_percent_to_sold") / 100))), 0).cast("int")) \
    .withColumn("potential_revenue", ((F.col("availability_items_count") + F.col("ordered_items_count")) * F.col("item_price")).cast("bigint")) \
    .withColumn("total_revenue", ((F.col("ordered_items_count") - F.col("returned_items_count") + F.col("goods_sold_count")) * F.col("item_price")).cast("bigint")) \
    .withColumn("avg_daily_sales", F.when(F.col("days_on_sell") > 0, F.col("goods_sold_count") / F.col("days_on_sell")).otherwise(0)) \
    .withColumn("days_to_sold", F.when(F.col("avg_daily_sales") > 0, F.col("availability_items_count") / F.col("avg_daily_sales")).otherwise(None))

# percent_rank по рейтингу
from pyspark.sql import Window
w = Window.orderBy("item_rate")
item_df = item_df.withColumn("item_rate_percent", F.percent_rank().over(w))

item_df.write.mode("overwrite").parquet(TARGET_PATH)

with DAG(dag_id="startde-project-sergej-vlasov-tnb4478-dag",
         schedule_interval=None, start_date=pendulum.datetime(2025,3,3)) as dag:
    
    submit_task = SparkKubernetesOperator(task_id="job_submit", application_file='items_spark_submit.yaml')
    sensor_task = SparkKubernetesSensor(task_id="job_sensor", application_name="{{ task_instance.xcom_pull(task_ids='job_submit')['metadata']['name'] }}")

-- VIEW: unreliable_sellers_view
CREATE VIEW "sergej-vlasov-tnb4478".unreliable_sellers_view AS
SELECT total.seller, total.availability_items_count AS total_overload_items_count,
       total.availability_items_count > total.ordered_items_count AS is_unreliable
FROM ( SELECT s.seller, SUM(s.availability_items_count) AS availability_items_count,
              SUM(s.ordered_items_count) AS ordered_items_count
       FROM "sergej-vlasov-tnb4478".seller_items s WHERE s.days_on_sell > 100
       GROUP BY s.seller ) AS total;

-- VIEW: item_brands_view
CREATE VIEW "sergej-vlasov-tnb4478".item_brands_view AS
SELECT brand, group_type, country,
       sum(potential_revenue) as potential_revenue,
       SUM(total_revenue) as total_revenue,
       count(sku_id) as items_count
FROM "sergej-vlasov-tnb4478".seller_items
GROUP BY brand, group_type, country;
